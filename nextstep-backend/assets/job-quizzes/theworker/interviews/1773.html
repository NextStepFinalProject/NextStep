  def my_xrange(start, end, steps=1):
if not steps:
raise ValueError("xrange() arg 3 must not be zero")
curr = start
if (start < end and steps > 0) or (start > end and steps < 0):
while curr < end or curr > end:
yield curr
curr += steps
