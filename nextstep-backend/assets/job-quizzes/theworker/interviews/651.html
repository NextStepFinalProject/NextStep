{
string sr = "";

public static string convertToString(int number,int count)
{
string c1 ="";
string c2 = "";
string c3 = "";

if(count == 1)
{
c1 = "I";
c2 = "V";
c3 = "X";
}
if (count == 10)
{
c1 = "X";
c2 = "L";
c3 = "C";
}
if(count ==100)
{
c1 = "C";
c2 = "D";
c3 = "M";
}
if(count == 1000)
{
c1 = "M";
}
string result="";
int n = number * count;
if(n*count == count)
{
result= c1;
}
if( 5 * count == number * count)
{
result= c2;
}
if (n <= 3 *count)
{
result= charXInt(c1, number);
}
if(n == 4 * count)
{
result = c1 + c2;
}
if(n > 5*count && n<=8*count)
{
string chars = charXInt(c1, number-5);
result = c2 + chars;
}
if(n == 9 *count)
{
result = c1 + c3;
}
return result;
}
