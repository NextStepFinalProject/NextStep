פרטים לגבי התהליך ראיון ראשון: עם מנהלת פיתוח בזום חצי שעה.
ראיון שני: עם ר"צ ועוד מתכנתת בזום. זה כלל גם שאלה ב sql.
שאלות מתוך הראיון השאלה:
יש 3 טבלאות הבאות: (לצורף הפתרון רק הטבלה האחרונה חשובה)

CREATE TABLE `Subscription` (
`ID` int(11) UNSIGNED NOT NULL,
`Plan_ID` int(11) UNSIGNED NOT NULL
)

CREATE TABLE `Databases` (
`ID` int(11) UNSIGNED NOT NULL,
`Subscription_ID` int(11) UNSIGNED NOT NULL
CONSTRAINT `subscription` FOREIGN KEY (`Subscription_ID`)
REFERENCES `Subscription`(`ID`)
)

CREATE TABLE `SubscriptionLog` (
`ID` int(11) UNSIGNED NOT NULL AUTO_INCREMENT,
`Subscription_ID` int(11) UNSIGNED NOT NULL,
`FromPlan_ID` int(11) UNSIGNED
`ToPlan_ID` int(11) UNSIGNED
`Changed` datetime
CONSTRAINT `subscription` FOREIGN KEY (`Subscription_ID`)
REFERENCES `Subscription`(`ID`)
)

צריך לתוב את השאילתה הבאה:
Select all months in the current year, with more than 5000 new subscriptions.
תשובות הוסף תשובה   |   לצפיה בתשובות
 
  דצמבר 2020
  זאת התשובה:

SELECT count(*) as num_of_subscriptions, MONTH(Changed) AS the_month
FROM SubscriptionLog
WHERE YEAR(Changed) = YEAR(CURDATE()) AND FromPlan_ID IS NULL
GROUP BY MONTH(Changed)
HAVING count(*) > 5000;
