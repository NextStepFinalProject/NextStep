public List FindItemsInBNotInA(List a, List b)
{
for (int i = 0; i < a.Count; i++)
{
int found = -1;

for (int j = 0; j < b.Count; j++)
{
if(a[i]==b[j])
{
found = j;
break;
}
}

if(found>0)
{
b.RemoveAt(found);
}
if(found==-1)
{
break;
}

}
return b;
}
