FROM (
SELECT CUSTOMER_ID,
AMOUNT,
UPDATE_DATE,
UPDATE_TIME,
ROW_NUMBER() OVER (PARTITION BY CUSTOMER_ID ORDER BY UPDATE_DATE, UPDATE_TIME DESC) AS RN
FROM CUSTOMER_DATA
) AS T
WHERE RN = 1
