using namespace std;

void printPermutations(vector elements, vector permutation)
{
if (elements.empty())
{
for (unsigned val : permutation)
cout << val << " ";

cout << endl;
}
else
for (size_t i = 0; i < elements.size(); i++)
{
unsigned temp = elements[i];
elements.erase(elements.begin() + i);
permutation.push_back(temp);
printPermutations(elements, permutation);
permutation.pop_back();
elements.insert(elements.begin() + i, temp);
}
}

void printPermutations(unsigned N)
{
vector elements, permutation;

for (unsigned i = 0; i < N; i++)
elements.push_back(i);

printPermutations(elements, permutation);
}
